clean_abund <- 
  abund %>%
  rowwise() %>%
  mutate(Sample_Sum = sum(c_across(contains(";"))),
         Sample = str_replace_all(Sample, " ", "_")) %>%
  select(Sample, Sample_Sum, Diet, Site, Day, mouse, Cage)


clean_together <- 
  clean_meta %>%
  rename(Sample = `Sample Name`) %>%
  inner_join(clean_abund, by = "Sample")
